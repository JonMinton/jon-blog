And to visualise

mtcars |> 
  mutate(cyl = factor(cyl)) |> 
  ggplot(aes(x = wt, y = mpg, colour = cyl, group= cyl)) + 
  geom_point(aes(shape = cyl)) + 
  stat_smooth(se = FALSE, method = "lm") + 
  labs(
    x = "Weight", 
    y = "Miles per gallon"
  )
